vec_y = currencies["growth_euro"]
plt.fill_between(x= currencies["date"],
                 y1 = vec_y.min(),
                 y2 = vec_y.max(),
                 where = currencies["bool_cond"],
                 alpha = 0.2,color = "red")

plt.show()

# Write your own code
month_config = pd.Grouper(key = "date",freq = "m")

monthly_data = currencies.groupby(month_config).agg(std_monthly = ("eur_to_usd","std")).reset_index()

plt.plot("date","std_monthly",
         data = monthly_data.sort_values("date",ascending = True))
plt.title("Volatitlity in the Exchange Between Euro to USD")
plt.xlabel("Time")
